% -- Representations
% A time phrase
\def\tp#1{\textit{#1}}
% A time expression
\def\te#1{\texttt{#1}}
% A time expression's type
\def\ty#1{\textsf{\small{#1}}}
% A dataset type
\def\dataset#1{#1}
% A system
\def\sys#1{#1}

% n^th of the month
\def\th#1{#1$^\textit{\small{th}}$}
% n^rd of the month
\def\rd#1{#1$^\textit{\small{rd}}$}
% n^nd of the month
\def\nd#1{#1$^\textit{\small{nd}}$}
