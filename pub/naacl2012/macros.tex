% -- Representations
% A time phrase
\def\tp#1{\textit{#1}}
% A time expression
\def\te#1{\texttt{#1}}
% A time expression's type
\def\ty#1{\textsf{\small{#1}}}
% A dataset type
\def\dataset#1{\texttt{#1}}
% -- Shortenings
% n^th of the month
\def\th#1{#1$^\textrm{th}$}
% n^rd of the month
\def\rd#1{#1$^\textrm{rd}$}
% n^nd of the month
\def\nd#1{#1$^\textrm{nd}$}
% -- Math Entities
\def\range{\ensuremath{r}}
\def\seq{\ensuremath{s}}
\def\seqelem{\ensuremath{r_s}}
\def\seqelemX#1{\ensuremath{r_{\te{\scriptsize{#1}}}}}
\def\seqjump{\ensuremath{\Delta_s}}
\def\seqjumpX#1{\ensuremath{\Delta_{\te{\scriptsize{#1}}}}}
\def\seqcanon{\ensuremath{\rho_s}}
\def\seqcanonX#1{\ensuremath{\rho_{\te{\scriptsize{#1}}}}}
\def\dur{\ensuremath{d}}
